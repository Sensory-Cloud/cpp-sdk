60struct CallbackData {
61 private:
63 ::grpc::ClientContext context;
65 ::grpc::Status status;
67 Request request;
69 Response response;
71 bool isDone = false;
73 std::mutex mutex;
75 std::condition_variable conditionVariable;

95 inline void setIsDone() {
96 // Lock the critical section for updating the `isDone` flag.
97 std::lock_guard<std::mutex> lock(mutex);
98 isDone = true;
99 // Notify the awaiting thread that the condition variable has changed.
100 conditionVariable.notify_one();
101 }

103 // Mark the Factory type as a friend to allow it to have write access to
104 // the internal types. This allows the parent scope to have mutability, but
105 // all other scopes must access data through the immutable `get` interface.
106 friend Factory;

108 public:

140 inline bool getIsDone() {
141 // Lock the critical section for querying the `isDone` flag.
142 std::lock_guard<std::mutex> lock(mutex);
143 return isDone;
144 }
145
152 inline void await() {
153 // Lock the critical section for updating the `isDone` flag and the
154 // gRPC `status` variable.
155 std::unique_lock<std::mutex> lock(mutex);
156 // Wait for the signal that the `isDone` flag has changed.
157 conditionVariable.wait(lock, [this] { return isDone; });
158 }
159};
