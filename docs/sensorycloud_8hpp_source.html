65static inline EnrollmentType parse_enrollment_type(const std::string& enrollment_type) {
66 if (enrollment_type == "none")
67 return EnrollmentType::None;
68 else if (enrollment_type == "sharedSecret")
69 return EnrollmentType::SharedSecret;
70 else if (enrollment_type == "jwt")
71 return EnrollmentType::JWT;
72 else
73 throw std::runtime_error("unrecognized enrollment type: " + enrollment_type);
74}

141template<typename CredentialStore>
142static inline std::string get_system_device_id(CredentialStore& keychain) {
143 if (keychain.contains(DEVICE_ID_KEYCHAIN_TAG))
144 return keychain.at(DEVICE_ID_KEYCHAIN_TAG);
145 auto device_id = sys::get_env_var(DEVICE_ID_ENV_KEY);
146 if (device_id.empty())
147 device_id = util::uuid_v4();
148 keychain.emplace(DEVICE_ID_KEYCHAIN_TAG, device_id);
149 return device_id;
150}

160template<typename CredentialStore>
161static inline std::string get_system_device_name(CredentialStore& keychain) {
162 if (keychain.contains(DEVICE_NAME_KEYCHAIN_TAG))
163 return keychain.at(DEVICE_NAME_KEYCHAIN_TAG);
164 auto name = sys::get_env_var(DEVICE_NAME_ENV_KEY);
165 if (name.empty())
166 name = util::uuid_v4();
167 keychain.emplace(DEVICE_NAME_KEYCHAIN_TAG, name);

169}

309 ::grpc::Status initialize(sensory::api::v1::management::DeviceResponse* response) {
310 if (token_manager.has_token()) return {}; // already registered.
311 const auto device_credentials = token_manager.has_saved_credentials() ?
312 token_manager.get_saved_credentials() : token_manager.generate_credentials();
313 // If we're using a JWT token, it needs to be signed.
314 auto credential = registration_credentials.credential;
315 std::string key = "-----BEGIN PRIVATE KEY-----\n" + credential + "\n-----END PRIVATE KEY-----";
316 if (registration_credentials.enrollment_type == EnrollmentType::JWT)
317 credential = jwt::create()
318 .set_issuer("sensorycloud-cpp-sdk")
319 .set_issued_at(std::chrono::system_clock::now())
320 // .set_expires_at(std::chrono::system_clock::now() + std::chrono::minutes{1})
321 .set_type("JWT")
322 .set_payload_claim("device_name", jwt::claim(registration_credentials.device_name))
323 .set_payload_claim("tenant_id", jwt::claim(token_manager.get_service().get_config().get_tenant_id()))
324 .set_payload_claim("client_id", jwt::claim(device_credentials.id))
325 .sign(jwt::algorithm::ed25519{"", key, "", ""});
326 auto status = oauth.register_device(response,
327 registration_credentials.device_name,
328 credential,
329 device_credentials.id,
330 device_credentials.secret
331 );
332 if (status.ok()) token_manager.get_access_token();
333 return status;
334 }
335};
