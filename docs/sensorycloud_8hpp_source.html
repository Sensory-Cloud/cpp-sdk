61static inline EnrollmentType parse_enrollment_type(const std::string& enrollment_type) {
62 if (enrollment_type == "none")
63 return EnrollmentType::None;
64 else if (enrollment_type == "sharedSecret")
65 return EnrollmentType::SharedSecret;
66 else if (enrollment_type == "jwt")
67 return EnrollmentType::JWT;
68 else
69 throw std::runtime_error("unrecognized enrollment type: " + enrollment_type);
70}

252 ::grpc::Status initialize(sensory::api::v1::management::DeviceResponse* response) {
253 if (token_manager.has_token()) return {}; // already registered.
254 const auto device_credentials = token_manager.has_saved_credentials() ?
255 token_manager.get_saved_credentials() : token_manager.generate_credentials();
256 // If we're using a JWT token, it needs to be signed.
257 auto credential = registration_credentials.credential;
258 std::string key = "-----BEGIN PRIVATE KEY-----\n" + credential + "\n-----END PRIVATE KEY-----";
259 if (registration_credentials.enrollment_type == EnrollmentType::JWT)
260 credential = jwt::create()
261 .set_issuer("sensorycloud-cpp-sdk")
262 .set_issued_at(std::chrono::system_clock::now())
263 // .set_expires_at(std::chrono::system_clock::now() + std::chrono::minutes{1})
264 .set_type("JWT")
265 .set_payload_claim("device_name", jwt::claim(registration_credentials.device_name))
266 .set_payload_claim("tenant_id", jwt::claim(token_manager.get_service().get_config().get_tenant_id()))
267 .set_payload_claim("client_id", jwt::claim(device_credentials.id))
268 .sign(jwt::algorithm::ed25519{"", key, "", ""});
269 return oauth.register_device(response,
270 registration_credentials.device_name,
271 credential,
272 device_credentials.id,
273 device_credentials.secret
274 );
275 }
