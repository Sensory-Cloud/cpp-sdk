45class TranscriptAggregator {
46 private:
48 std::vector<::sensory::api::v1::audio::TranscribeWord> word_list;
49
50 public:
55 void process_response(const ::sensory::api::v1::audio::TranscribeWordResponse& response) {
56 if (response.words().empty()) return;
57 // Get the expected transcript size from the index of the last word.
58 const auto response_size = response.lastwordindex() + 1;
59 // Grow the word buffer if the incoming transcript is larger.
60 if (response_size > word_list.size())
61 word_list.resize(response_size);
62 // Loop through returned words and replace buffered words that changed.
63 for (const auto& word : response.words()) {
64 // Sanity check the word index to prevent the possibility of
65 // unexpected segmentation faults in favor of descriptive errors.
66 if (word.wordindex() >= word_list.size())
67 throw std::runtime_error(
68 "Attempting to update word at index " +
69 std::to_string(word.wordindex()) +
70 " that exceeds the expected buffer size of " +
71 std::to_string(word_list.size())
72 );
73 word_list[word.wordindex()] = word;
74 }
75 // Shrink the word list if the incoming transcript is smaller.
76 if (response_size < word_list.size())
77 word_list.erase(word_list.begin() + response_size, word_list.end());
78 }
79
84 inline const std::vector<::sensory::api::v1::audio::TranscribeWord>& get_word_list() const {
85 return word_list;
86 }
87
94 std::string get_transcript(const std::string& delimiter=" ") const {
95 if (word_list.empty()) return "";
96 // Iterate over the words to accumulate the transcript.
97 std::string transcript = "";
98 for (const auto& word : word_list)
99 transcript += delimiter + ::sensory::util::strip(word.word());
100 // Remove the extra space at the front of the transcript.
101 transcript.erase(0, 1);
102 return transcript;
103 }
104};
