39struct ConfigError : public std::runtime_error {

42 enum class Code {

48 InvalidPort,

60 static inline const std::string get_message(const Code& code) {
61 switch (code) { // switch over the possible code type cases
62 case Code::InvalidFQDN:
63 return "The fully qualified domain name is not valid";
64 case Code::InvalidHost:
65 return "The host name is not valid";
66 case Code::InvalidPort:
67 return "The port number is not valid";
68 case Code::InvalidTenantID:
69 return "The tenant ID is not valid";
70 case Code::InvalidDeviceID:
71 return "The device ID is not valid";
72 default:
73 return "Unrecognized error code";
74 }
75 }
76
81 explicit ConfigError(const Code& code) :
82 std::runtime_error(get_message(code)),
83 err_code(code) { }
84
90 explicit ConfigError(const Code& code, const std::string& message) :
91 std::runtime_error(message),
92 err_code(code) { }
93
95 ~ConfigError() throw() {}
96
101 inline const Code& code() const throw() { return err_code; }
102
103 private:
105 Code err_code;
