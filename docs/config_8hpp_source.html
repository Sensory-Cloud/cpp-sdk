76 Config(
77 const std::string& fqdn_,
78 const std::string& tenant_id_,
79 const std::string& device_id_,
80 const bool& is_secure_ = true
81 ) :
82 fqdn(fqdn_),
83 tenant_id(tenant_id_),
84 device_id(device_id_),
85 is_secure(is_secure_) {
86 // Check that the FQDN is properly formatted in `host:port` format.
87 const auto idx = fqdn.find(':');
88 if (fqdn.empty() || idx == 0 || idx >= fqdn.length() - 1)
89 throw ::sensory::error::ConfigError(::sensory::error::ConfigError::Code::InvalidFQDN);
90 // Parse the port as a 32-bit signed integer and ensure that the value
91 // is a valid 16-bit unsigned integer.
92 const auto port = std::stoi(fqdn.substr(idx + 1));
93 if (port < std::numeric_limits<uint16_t>::min() ||
94 port > std::numeric_limits<uint16_t>::max())
95 throw ::sensory::error::ConfigError(::sensory::error::ConfigError::Code::InvalidPort);
96 // Ensure the tenant ID and device ID are not empty.
97 if (tenant_id.empty()) // the tenant ID is not valid
98 throw ::sensory::error::ConfigError(::sensory::error::ConfigError::Code::InvalidTenantID);
99 if (device_id.empty()) // the device ID is not valid
100 throw ::sensory::error::ConfigError(::sensory::error::ConfigError::Code::InvalidDeviceID);
101 // Create the credentials for the channel based on the security setting.
102 // Use TLS (SSL) if `is_secure` is true, otherwise default to insecure
103 // channel credentials.
104 channel = ::grpc::CreateChannel(fqdn, is_secure ?
105 ::grpc::SslCredentials(::grpc::SslCredentialsOptions()) :
106 ::grpc::InsecureChannelCredentials()
107 );
108 }

151 inline std::string get_host() const {
152 // Note: Typically one would need to check that `find` returned a value
153 // other than std::string::npos; however, the fully qualified domain
154 // name is immutable and the existence of the ':' character is
155 // guaranteed past initialization time.

157 }

163 inline uint16_t get_port() const {
164 // Note: Typically one would need to check that `find` returned a value
165 // other than std::string::npos; however, the fully qualified domain
166 // name is immutable and the existence of the ':' character is
167 // guaranteed past initialization time.
168 return std::stoi(fqdn.substr(fqdn.find(':') + 1));
169 }
