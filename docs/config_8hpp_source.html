76 Config(
77 const std::string& fqdn_,
78 const std::string& tenant_id_,
79 const std::string& device_id_,
80 const bool& is_secure_ = true
81 ) :
82 fqdn(fqdn_),
83 tenant_id(tenant_id_),
84 device_id(device_id_),
85 is_secure(is_secure_) {
86 if (fqdn.empty()) // The FQDN is not valid
87 throw ::sensory::error::ConfigError(::sensory::error::ConfigError::Code::InvalidFQDN);
88 if (tenant_id.empty()) // the tenant ID is not valid
89 throw ::sensory::error::ConfigError(::sensory::error::ConfigError::Code::InvalidTenantID);
90 if (device_id.empty()) // the device ID is not valid
91 throw ::sensory::error::ConfigError(::sensory::error::ConfigError::Code::InvalidDeviceID);
92 // Create the credentials for the channel based on the security setting.
93 // Use TLS (SSL) if `is_secure` is true, otherwise default to insecure
94 // channel credentials.
95 channel = ::grpc::CreateChannel(fqdn, is_secure ?
96 ::grpc::SslCredentials(::grpc::SslCredentialsOptions()) :
97 ::grpc::InsecureChannelCredentials()
98 );
99 }
