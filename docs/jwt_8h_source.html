398 inline std::string extract_pubkey_from_cert(const std::string& certstr, const std::string& pw,
399 std::error_code& ec) {
400 ec.clear();
401#if OPENSSL_VERSION_NUMBER <= 0x10100003L
402 std::unique_ptr<BIO, decltype(&BIO_free_all)> certbio(
403 BIO_new_mem_buf(const_cast<char*>(certstr.data()), static_cast<int>(certstr.size())), BIO_free_all);
404#else
405 std::unique_ptr<BIO, decltype(&BIO_free_all)> certbio(
406 BIO_new_mem_buf(certstr.data(), static_cast<int>(certstr.size())), BIO_free_all);
407#endif
408 std::unique_ptr<BIO, decltype(&BIO_free_all)> keybio(BIO_new(BIO_s_mem()), BIO_free_all);
409 if (!certbio || !keybio) {
410 ec = error::rsa_error::create_mem_bio_failed;
411 return {};
412 }
413
414 std::unique_ptr<X509, decltype(&X509_free)> cert(
415 PEM_read_bio_X509(certbio.get(), nullptr, nullptr, const_cast<char*>(pw.c_str())), X509_free);
416 if (!cert) {
417 ec = error::rsa_error::cert_load_failed;
418 return {};
419 }
420 std::unique_ptr<EVP_PKEY, decltype(&EVP_PKEY_free)> key(X509_get_pubkey(cert.get()), EVP_PKEY_free);
421 if (!key) {
422 ec = error::rsa_error::get_key_failed;
423 return {};
424 }
425 if (PEM_write_bio_PUBKEY(keybio.get(), key.get()) == 0) {
426 ec = error::rsa_error::write_key_failed;
427 return {};
428 }
429 char* ptr = nullptr;
430 auto len = BIO_get_mem_data(keybio.get(), &ptr);
431 if (len <= 0 || ptr == nullptr) {
432 ec = error::rsa_error::convert_to_pem_failed;
433 return {};
434 }
435 return {ptr, static_cast<size_t>(len)};
436 }

468 std::error_code& ec) {
469 ec.clear();
470 const auto decodedStr = decode(cert_base64_der_str);
471 auto c_str = reinterpret_cast<const unsigned char*>(decodedStr.c_str());
472
473 std::unique_ptr<X509, decltype(&X509_free)> cert(
474 d2i_X509(NULL, &c_str, static_cast<int>(decodedStr.size())), X509_free);
475 std::unique_ptr<BIO, decltype(&BIO_free_all)> certbio(BIO_new(BIO_s_mem()), BIO_free_all);
476 if (!cert || !certbio) {

480

482 ec = error::rsa_error::write_cert_failed;
483 return {};
484 }
485
486 char* ptr = nullptr;
487 const auto len = BIO_get_mem_data(certbio.get(), &ptr);
488 if (len <= 0 || ptr == nullptr) {
489 ec = error::rsa_error::convert_to_pem_failed;
490 return {};
491 }
492
493 return {ptr, static_cast<size_t>(len)};
494 }

559 inline std::shared_ptr<EVP_PKEY> load_public_key_from_string(const std::string& key,
560 const std::string& password, std::error_code& ec) {
561 ec.clear();
562 std::unique_ptr<BIO, decltype(&BIO_free_all)> pubkey_bio(BIO_new(BIO_s_mem()), BIO_free_all);
563 if (!pubkey_bio) {
564 ec = error::rsa_error::create_mem_bio_failed;
565 return nullptr;
566 }
567 if (key.substr(0, 27) == "-----BEGIN CERTIFICATE-----") {
568 auto epkey = helper::extract_pubkey_from_cert(key, password, ec);
569 if (ec) return nullptr;
570 const int len = static_cast<int>(epkey.size());
571 if (BIO_write(pubkey_bio.get(), epkey.data(), len) != len) {
572 ec = error::rsa_error::load_key_bio_write;
573 return nullptr;
574 }
575 } else {
576 const int len = static_cast<int>(key.size());
577 if (BIO_write(pubkey_bio.get(), key.data(), len) != len) {
578 ec = error::rsa_error::load_key_bio_write;
579 return nullptr;
580 }
581 }
582
583 std::shared_ptr<EVP_PKEY> pkey(
584 PEM_read_bio_PUBKEY(pubkey_bio.get(), nullptr, nullptr,
585 (void*)password.data()), // NOLINT(google-readability-casting) requires `const_cast`
586 EVP_PKEY_free);
587 if (!pkey) {
588 ec = error::rsa_error::load_key_bio_read;
589 return nullptr;
590 }
591 return pkey;
