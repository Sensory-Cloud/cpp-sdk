140 // Execute the RPC synchronously and return the status
141 return device_stub->EnrollDevice(&context, request, response);

175 inline RegisterDeviceAsyncCall* register_device(
176 ::grpc::CompletionQueue* queue,
177 const std::string& name,
178 const std::string& credential,
179 const std::string& client_id,
180 const std::string& client_secret
181 ) const {
182 // Create a call data object to store the client context, the response,
183 // the status of the call, and the response reader. The ownership of
184 // this object is passed to the caller.
185 auto call(new RegisterDeviceAsyncCall);
186 // Start the asynchronous RPC with the call's context and queue.
187 call->request.set_deviceid(config.get_device_id());
188 call->request.set_tenantid(config.get_tenant_id());
189 call->request.set_name(name);
190 call->request.set_credential(credential);
191 auto client_request = new ::sensory::api::common::GenericClient;
192 client_request->set_clientid(client_id);
193 client_request->set_secret(client_secret);
194 call->request.set_allocated_client(client_request);
195 call->rpc = device_stub->AsyncEnrollDevice(&call->context, call->request, queue);
196 // Finish the RPC to tell it where the response and status buffers are
197 // located within the call object. Use the address of the call as the
198 // tag for identifying the call in the event-loop.
199 call->rpc->Finish(&call->response, &call->status, static_cast<void*>(call));
200 // Return the pointer to the call. This both transfers the ownership
201 // of the instance to the caller, and provides the caller with an
202 // identifier for detecting the result of this call in the completion
203 // queue.
204 return call;
205 }
