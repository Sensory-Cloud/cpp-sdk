58template<typename Factory, typename Request>
59class AwaitableWriteReactor : public ::grpc::ClientWriteReactor<Request> {
60 private:
62 ::grpc::ClientContext context;
64 ::grpc::Status status;
66 bool isDone;
68 std::mutex mutex;
70 std::condition_variable conditionVariable;
71
72 // Mark the Factory type as a friend to allow it to have write access to
73 // the internal types. This allows the parent scope to have mutability, but
74 // all other scopes must access data through the immutable `get` interface.
75 friend Factory;
76
77 public:
79 Request request;
80
82 AwaitableWriteReactor() : isDone(false) { }
83
88 inline virtual void OnDone(const grpc::Status& status_) override {
89 // Lock the critical section for updating the `isDone` flag and the
90 // gRPC `status` variable.
91 std::lock_guard<std::mutex> lock(mutex);
92 status = status_;
93 isDone = true;
94 // Notify the awaiting thread that the condition variable has changed.
95 conditionVariable.notify_one();
96 }
97
102 inline ::grpc::Status getStatus() {
103 // Lock the critical section for querying the `status`.
104 std::lock_guard<std::mutex> lock(mutex);
105 return status;
106 }
107
112 inline bool getIsDone() {
113 // Lock the critical section for querying the `isDone` flag.
114 std::lock_guard<std::mutex> lock(mutex);
115 return isDone;
116 }
117
122 inline ::grpc::Status await() {
123 // Lock the critical section for updating the `isDone` flag and the
124 // gRPC `status` variable.
125 std::unique_lock<std::mutex> lock(mutex);
126 // Wait for the signal that the `isDone` flag has changed.
127 conditionVariable.wait(lock, [this] { return isDone; });
128 return status;
129 }
130};
