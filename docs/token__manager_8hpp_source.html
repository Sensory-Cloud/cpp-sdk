117 inline AccessTokenCredentials generate_credentials() const {
118 // Generate a new client ID and secure random secret string.
119 const auto clientID = ::sensory::util::uuid_v4(); // v4 UUIDs don't identify the host.
120 const auto secret = ::sensory::util::secure_random(24); // Use a 24 character secret.
121 // Insert the clientID and secret into the persistent credential store.
122 // If any key-value pair already exists, overwrite it.
123 credential_store.emplace(TAGS.ClientID, clientID);
124 credential_store.emplace(TAGS.ClientSecret, secret);
125 // Return a new access token with the credentials
126 return {clientID, secret};
127 }

134 inline AccessTokenCredentials get_saved_credentials() const {
135 return {credential_store.at(TAGS.ClientID), credential_store.at(TAGS.ClientSecret)};
136 }

189 std::string get_access_token() const {
190 // Prevent multiple access tokens from being requested at the same time.
191 static std::mutex mutex;
192 std::lock_guard<std::mutex> lock(mutex);
193
194 if (!has_token()) // no access token has been generated and stored
195 return fetch_new_access_token();
196 // fetch existing access token and expiration date from the secure store
197 const auto access_token = credential_store.at(TAGS.AccessToken);
198 const auto expiration_date = credential_store.at(TAGS.Expiration);
199 // check for expiration of the token
200 const auto now = std::chrono::system_clock::now();
201 const auto expiration = ::sensory::util::timestamp_to_timepoint(expiration_date);
202 if (now > expiration - std::chrono::minutes(5)) // token has expired
203 return fetch_new_access_token();
204 return access_token;
205 }

211 std::string fetch_new_access_token() const {
212 // Check if credentials have been generated already, otherwise create
213 // new credentials.
214 if (!has_saved_credentials()) generate_credentials();
215 // Get the ID of the client and the secret from the secure store.
216 const auto client_id = credential_store.at(TAGS.ClientID);
217 const auto secret = credential_store.at(TAGS.ClientSecret);
218 // Synchronously request a new token from the server.
219 api::common::TokenResponse response;
220 const auto status = service.get_token(&response, client_id, secret);
221 // Insert the OAuth access token for the client in the secure store
222 credential_store.emplace(TAGS.AccessToken, response.accesstoken());
223 // Determine when the token will expire and store this time
224 const auto expiration_date =
225 std::chrono::system_clock::now() + std::chrono::seconds(response.expiresin());
226 credential_store.emplace(TAGS.Expiration, ::sensory::util::timepoint_to_timestamp(expiration_date));
227 // Return the newly created OAuth token
228 return response.accesstoken();
229 }
