45inline std::string timepoint_to_timestamp(const std::chrono::system_clock::time_point& time_point) {
46 // Convert the time point to a time_t object
47 const auto tt = std::chrono::system_clock::to_time_t(time_point);
48 // Format the time point into a character buffer (as a string).
49 std::string output("0000-00-00T00:00:00Z");
50 // use a local `std::tm` buffer and the accompanying `gmtime_r` function
51 // instead of the static memory used by `gmtime`. This is for thread safety.
52 std::tm tm;
53 strftime(&output[0], output.length(), "%Y-%m-%dT%H:%M:%SZ", gmtime_r(&tt, &tm));
54 return std::move(output);
55}

62inline std::chrono::system_clock::time_point timestamp_to_timepoint(const std::string& timestamp) {
63 // Stream the contents of the timestamp string into the tm structure
64 std::tm tm = {};
65 strptime(timestamp.c_str(), "%Y-%m-%dT%H:%M:%SZ", &tm);
66 // Convert tm -> time_t in GMT format -> time_point
67 return std::chrono::system_clock::from_time_t(timegm(&tm));
68}
